function prosesSimulasi() {
    var inputUser = document.getElementById("username").value;
    
    // Validasi sederhana agar input tidak kosong saat demo

        alert("Silakan masukkan ID sampel untuk melihat cara kerja simulasi ini.");
        return;
    }
    
    // Menampilkan data input secara lokal di browser penonton (Client-side saja, 100% Aman)
    document.getElementById("data-terinput").innerText = inputUser;
    
    // Transisi menyembunyikan form dan menampilkan materi edukasi
    document.getElementById("panel-login").style.display = "none";
    document.getElementById("panel-edukasi").style.display = "block";
}
